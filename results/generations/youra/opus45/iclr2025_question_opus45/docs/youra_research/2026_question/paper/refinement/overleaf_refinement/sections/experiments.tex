
\subsubsection{4.1 Research Questions}

\textbf{RQ1 (Existence Proof):} Does SEDP achieve Spearman rho >= 0.3 with true semantic entropy?

\textbf{RQ2 (Similarity Benefit):} Does SEDP (hidden + similarity) outperform SEP (hidden only)?

\subsubsection{4.3 Model Configuration}

\textbf{Language Model:} Llama-3-8B-Instruct (meta-llama/Meta-Llama-3-8B-Instruct)
\begin{itemize}
\item Architecture: Decoder-only transformer, 32 layers
\item Hidden dimension: 4096
\item Precision: float16
\end{itemize}

\textbf{Generation Settings:}
\begin{itemize}
\item Responses per question: N = 20
\item Temperature: T = 0.7
\item Max new tokens: 100
\end{itemize}

\textbf{Hidden State Extraction:}
\begin{itemize}
\item Layer: 25 of 32
\item Token position: TBG (Token Before Generation)
\end{itemize}

\textbf{Entailment Model:} DeBERTa-v3-large-MNLI
\begin{itemize}
\item Entailment threshold: 0.5 (bidirectional)
\end{itemize}

\textbf{Similarity Embedding:} all-MiniLM-L6-v2

\subsubsection{4.4 Baselines}

\textbf{SEP (Semantic Entropy Probe):} Logistic regression on hidden states only (4096 features).

\textbf{Random Baseline:} AUROC = 0.50, Spearman rho = 0.0.

\subsubsection{4.5 Implementation}

\begin{itemize}
\item Probe: sklearn LogisticRegression, C=1.0, LBFGS optimizer, max\_iter=1000
\item Hardware: Single NVIDIA A100 (40GB)
\item Response generation: approximately 2-4 hours
\item SE label computation: approximately 1 hour
\item Probe training: under 1 minute
\end{itemize}
