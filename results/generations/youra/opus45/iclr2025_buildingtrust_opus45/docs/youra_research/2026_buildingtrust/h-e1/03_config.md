### Experiment Configuration Dataclass

```python
from dataclasses import dataclass, field
from typing import Dict, Tuple
import torch

@dataclass
class ExperimentConfig:
    """Main configuration for H-E1 AUROC experiment."""
    seed: int = 42
    bootstrap_n: int = 1000
    checkpoint_interval: int = 1000
    dtype: torch.dtype = torch.float16
    device: str = "cuda"
    results_dir: str = "results"
    figures_dir: str = "figures"
    cache_dir: str = "cache"
    dataset_name: str = "cais/mmlu"
    dataset_config: str = "all"
    dataset_split: str = "test"
    temperature: float = 0.0

@dataclass
class ModelPairsConfig:
    """Registry of all model pairs for comparison."""
    pairs: Dict[str, Tuple[str, str]] = field(default_factory=lambda: {
        "qwen": ("Qwen/Qwen2.5-7B", "Qwen/Qwen2.5-7B-Instruct"),
        "llama": ("meta-llama/Llama-2-7b-hf", "meta-llama/Llama-2-7b-chat-hf"),
        "mistral": ("mistralai/Mistral-7B-v0.1", "mistralai/Mistral-7B-Instruct-v0.2"),
    })

@dataclass
class PromptConfig:
    """Configuration for MCQ prompt formatting."""
    template: str = "Question: {question}\nA. {a}\nB. {b}\nC. {c}\nD. {d}\nAnswer:"
    choice_labels: Tuple[str, str, str, str] = ("A", "B", "C", "D")
    choice_token_prefix: str = " "

@dataclass
class GateConfig:
    """Configuration for MUST_WORK gate evaluation."""
    gate_type: str = "MUST_WORK"
    require_positive_delta: bool = True
    require_ci_excludes_zero: bool = True
    require_all_families: bool = True
```

## YAML Configuration File

```yaml
# config.yaml - H-E1 Experiment Configuration
experiment:
  name: "H-E1 AUROC Discriminative Degradation"
  hypothesis_id: "h-e1"
  hypothesis_type: "EXISTENCE"

reproducibility:
  seed: 42

bootstrap:
  n_iterations: 1000
  ci_percentiles: [2.5, 97.5]

checkpointing:
  interval: 1000
  enabled: true

model:
  dtype: "float16"
  device: "cuda"

models:
  qwen:
    base: "Qwen/Qwen2.5-7B"
    instruct: "Qwen/Qwen2.5-7B-Instruct"
  llama:
    base: "meta-llama/Llama-2-7b-hf"
    instruct: "meta-llama/Llama-2-7b-chat-hf"
  mistral:
    base: "mistralai/Mistral-7B-v0.1"
    instruct: "mistralai/Mistral-7B-Instruct-v0.2"

dataset:
  name: "cais/mmlu"
  config: "all"
  split: "test"

inference:
  temperature: 0.0
  max_new_tokens: 1

paths:
  results: "results"
  figures: "figures"
  cache: "cache"

gate:
  type: "MUST_WORK"
  criteria:
    - "AUROC_base > AUROC_instruct for all families"
    - "95% CI of delta excludes zero"
```

## Default Values and Justification

| Parameter | Default | Justification |

| seed | 42 | Standard reproducibility seed |

| checkpoint_interval | 1000 | Balance I/O overhead and recovery |
| dtype | float16 | Fits 7B model in ~14GB VRAM |
| device | "cuda" | Single GPU execution per PRD |
| temperature | 0.0 | Greedy decoding for determinism |
